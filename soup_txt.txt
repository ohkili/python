	// 원스탑 화면 리사이징 - 화면 크기 모드 (S:900*587, L:900*682)
	function fnBookMainResize(lv){
		var totalHeight = 0;
		var resizeH = 0;

		if( sBrowser.msie ){ //IE
			totalHeight = document.body.clientHeight;
		} else if ( sBrowser.firefox || sBrowser.opera || sBrowser.safari ){ // Firefox || NS, Opera, Chrome || Safari
			totalHeight = window.innerHeight;
		} else {      
			totalHeight = 682;
		}   

		if (lv == "L"){
			resizeH = 682 - totalHeight;
		}else{
			resizeH = 588 - totalHeight;
		}

		if (resizeH != 0){
			window.resizeBy(0, resizeH);
			//alert(document.body.clientHeight);
		}
	}

	// iframe 페이지 submit
	function fnIframeSubmit(formName, iframeName, actionUrl){
		var form = $(formName);
		
		form.target = iframeName;
		form.action = actionUrl;
		form.submit();
		
		var iframe = $(iframeName);
		Event.observe(iframe, "load", function(){
			try{
				// iframe에 각 단계 별 화면이 load 되지 않고 
				// loading.html이나 ticketImage.interpark.com/loading.html 화면이 렌더링 되있을 경우 
				// 재 submit 처리 진행.
				if(iframe.contentWindow.document.location.pathname.indexOf('loading.html') != -1){
					form.submit();	
				}
			}catch(e){
				// contentWindow의 document 접근 시 CORS 이슈 발생.
				// 재 submit 처리 진행.
				form.submit();
			}
		});
	}

	// 네비 변경
	function fnNaviChange(){

		var objNaviLink = $N("NaviLink");
		var objNaviImg = $N("NaviImg");
		var naviImg = "http://ticketimage.interpark.com/TicketImage/onestop/0";
		var naviCnt = objNaviImg.length;
		
		for (var i = 0; i < naviCnt; i++){
			var bookStep = objNaviLink[i].getAttribute("BookStep");

			var tempNaviImg = naviImg + naviCnt + "_step_0" + (i + 1);
			
			// 네비 이미지 포커스 변경
			if (bookStep == nNowBookStep || (bookStep == 5 && nNowBookStep == 6)){
				objNaviImg[i].src = tempNaviImg + "_on.gif";
			}else{
				objNaviImg[i].src = tempNaviImg + "_off.gif";
			}
			
			// 네비 링크 변경(6단계-카트입력완료 에서는 링크 비활성화)
			if (bookStep < nNowBookStep && nNowBookStep != 6){
				objNaviLink[i].style.cursor = "";
				objNaviLink[i].href = fnGetNaviLinkFunction(bookStep);
				//이전단계 이동 가능 이미지로 변경
				objNaviImg[i].src = tempNaviImg + "_back.gif";
				objNaviImg[i].title = objNaviImg[i].alt + " 단계로 이동";
			}else{
				objNaviLink[i].style.cursor = "default";
				objNaviLink[i].href = "#";

				objNaviImg[i].title = "";
			}
		}

	}
	
	// 네비 링크 조회
	function fnGetNaviLinkFunction(nNaviStep){
		var functionName = "#";
		if (nNaviStep == 1){
			functionName = "javascript:fnOpenBookDatatime();";	// 관람일&회차선택
		}else if (nNaviStep == 2){
			functionName = "javascript:fnOpenBookSeat();";	// 좌석선택
		}else if (nNaviStep == 3){
			functionName = "javascript:fnOpenBookPriceDiscount();";	// 가격/할인선택
		}else if (nNaviStep == 4){
			functionName = "javascript:fnOpenBookDelivery();";	//배송선택/주문자확인
		}
		return functionName;
	}

	// 메인 화면 갱신
	//////////////////////////////////////////////////////////////////
	function fnUIRefresh(){
		//예매일시

		var sPlayDateValue = $F("PlayDate");
		var sPlayTimeValue = $F("PlayTime");

//		if (sPlayTimeValue == "상시상품"){
//			sPlayTimeValue = "";
//		}

		if (sPlayDateValue != ""){
			if (sPlayTimeValue != ""){
				//미래세움 종일권
				if( ("21004791" == "14011965" || "21004791" == "14014183" || "21004791" == "16002098") && sPlayTimeValue == "2100"){
					$("MyPlayDate").innerHTML = fnKoreanDateMode(sPlayDateValue,1) + " <br />종일권";
				}else{
					$("MyPlayDate").innerHTML = fnKoreanDateMode(sPlayDateValue, 1) + " " + fnTimeFomat(sPlayTimeValue);
				}
			}else{
				$("MyPlayDate").innerHTML = fnKoreanDateMode(sPlayDateValue, 1);
			}
		}else{
			$("MyPlayDate").innerHTML = "";
		}

		$("MyPlayDate").title = $("MyPlayDate").innerHTML;
		
		//선택좌석 및 매수, 티켓금액 & 수수료 계산	
		var str = "";
		var cnt = 0;
		var amt = 0;
		var uamt = 0;
		var vamt = 0;		
		var tmpFloor, tmpRowNo, tmpSeatNo;
		var tempPreSeatGradeName = "";

		for(var i = 0; i < $N("SeatGrade").length; i++){
			tmpFloor = fnTrim($N("Floor")[i].value);
			tmpRowNo = fnTrim($N("RowNo")[i].value);
			tmpSeatNo = fnTrim($N("SeatNo")[i].value);

			if(fnTrim($N("SeatGrade")[i].value) != ""){
				
					str += "<li>";
					str += $N("SeatGradeName")[i].value;
					if ($N("SeatInfo")[i].value != ""){
						str += $N("SeatInfo")[i].value;
					}else{
						str += (tmpFloor == "" ? "" : " " + tmpFloor);
						str += (tmpRowNo == "" ? "" : " " + tmpRowNo);
						str += (tmpSeatNo == "" ? "" : " " + tmpSeatNo);
					}
					str += "</li>";
				
				
				cnt++;
				
				amt += eval($N("SalesPrice")[i].value);
				
				if(fnTrim($N("VoucherCode")[i].value) == ""){

					uamt += 500;
				
				}else{
					vamt += eval($N("SalesPrice")[i].value);
				}
			}
		}

		//휴대폰 수수료 계산은 결제하기 단계에서 실행
		if (nNowBookStep >= 5){
			uamt += eval($F("MCash_UseAmt"));
		}

		$("TicketCnt").value = cnt;
		$("TicketAmt").value = amt;
		$("TotalUseAmt").value = uamt;
		$("VoucherAmt").value = vamt;
		
		$("MySelectedSeatCnt").innerHTML = cnt;
		$("MySelectedSeat").innerHTML = str;

		//배송료 텍스트 변경
		var nDelyFee = 2800;
		var nExpressDelyFee = 0;



		var MyDeliveryAmt = "";
		if($("DeliveryOrNot").value != ""){
			var DeliveryGiftName = "";
			DeliveryGiftName = GetDeliveryGiftName($F("DeliveryGift"));
			if($F("Delivery") == "24001"){
				if(fnTrim($F("DeliveryGift")) != ""){
					//$("DelyAmt").value = parseInt(2800) + parseInt($F("DeliveryGiftAmt"));
					$("DelyAmt").value = parseInt(nDelyFee) + parseInt($F("DeliveryGiftAmt"));
					MyDeliveryAmt = fnAddComma($F("DelyAmt")) + "원 | " + DeliveryGiftName + " 배송";
				}else{
					//$("DelyAmt").value = "2800";
					$("DelyAmt").value = nDelyFee;
					MyDeliveryAmt = fnAddComma($F("DelyAmt")) + "원 | 배송";
				}
			}else if($F("Delivery") == "24004"){
				if(fnTrim($F("DeliveryGift")) != ""){
					//$("DelyAmt").value = 0 + parseInt($F("DeliveryGiftAmt"));
					$("DelyAmt").value = nExpressDelyFee + parseInt($F("DeliveryGiftAmt"));
					MyDeliveryAmt = fnAddComma($F("DelyAmt")) + "원 | " + DeliveryGiftName + " 특급배송";
				}else{
					//$("DelyAmt").value = "0";
					$("DelyAmt").value = nExpressDelyFee;
					MyDeliveryAmt = fnAddComma($F("DelyAmt")) + "원 | 특급배송";
				}
			}else if($F("Delivery") == "24011"){
				$("DelyAmt").value = "0";
				MyDeliveryAmt = fnAddComma($F("DelyAmt")) + "원 | 모바일티켓";
			}else if($F("Delivery") == "24012"){
				$("DelyAmt").value = "0";
				MyDeliveryAmt = fnAddComma($F("DelyAmt")) + "원 | 홈티켓";
			}else if($F("Delivery") == "24016"){
				$("DelyAmt").value = "0";
				MyDeliveryAmt = fnAddComma($F("DelyAmt")) + "원 | 모바일티켓";
			}else{
				$("DelyAmt").value = "0";
				MyDeliveryAmt = fnAddComma($F("DelyAmt")) + "원 | 현장수령";
			}
		}else{
			$("DelyAmt").value = "0";
			$("MyDeliveryAmt").innerHTML = fnAddComma($F("DelyAmt")) + "원";
		}

		//티켓금액, 수수료, 배송료는 가격선택 단계부터 노출
		if (nNowBookStep >= 3){
			$("MyTicketAmt").innerHTML = fnAddComma(eval($F("TicketAmt")) - eval($F("VoucherAmt"))) + "원";
			$("MyTicketAmt").title = $("MyTicketAmt").innerHTML;
			$("MyUseAmt").innerHTML = fnAddComma($F("TotalUseAmt")) + "원";
			$("MyUseAmt").title = $("MyUseAmt").innerHTML;
			$("MyDeliveryAmt").innerHTML = MyDeliveryAmt;
			$("MyDeliveryAmt").title = MyDeliveryAmt;
		}else{
			$("MyTicketAmt").innerHTML = "";
			$("MyTicketAmt").title = "";
			$("MyUseAmt").innerHTML = "";
			$("MyUseAmt").title = "";
			$("MyDeliveryAmt").innerHTML = "";
			$("MyDeliveryAmt").title = "";
		}

		//할인 내역 표시
		var disdesc = "";
		if(eval($F("DiscountAmt")) > 0 && eval($F("TicketAmt")) - eval($F("DiscountAmt")) > 0){
			disdesc = fnAddComma(eval($F("TicketAmt")) - eval($F("DiscountAmt"))) + "원 | " + $F("DiscountName");
		}else{
			disdesc = $F("DiscountName");
		}
		if($F("Point") == "Y" && eval($F("PointDiscountAmt")) > 0 && $F("SecondKindOfPayment") != "22060"){
			if(disdesc != ""){
				disdesc += " / " + fnAddComma($F("PointDiscountAmt")) + "P | 아이포인트 사용";
			}else{
				disdesc = fnAddComma($F("PointDiscountAmt")) + "P | 아이포인트 사용";
			}
		}
		if($F("OKCashbagYN") == "Y" && eval($F("OKCashbagPoint")) > 0){
		    if(disdesc != ""){
		        disdesc += " / " + fnAddComma($F("OKCashbagPoint")) + "P | 아이포인트 사용";
		    }else{
		        disdesc = fnAddComma($F("OKCashbagPoint")) + "P | OK캐쉬백 사용";
		    }
		}

		if($F("CareerYN") == "Y" && eval($F("CareerPoint")) > 0)
		{
		    if(disdesc != "")
		    {
		        disdesc += " / " + fnAddComma($F("CareerPoint")) + "P | 아이포인트 사용";
		    }
		    else
		    {
		        disdesc = fnAddComma($F("CareerPoint")) + "P | 바우처카드 포인트 사용";
		    }
		}
		if($F("CJOneYN") == "Y" && eval($F("CJOnePoint")) > 0){
			if(disdesc != ""){
				disdesc += " / " + fnAddComma($F("CJOnePoint")) + "P | 아이포인트 사용";
			}else{
				disdesc = fnAddComma($F("CJOnePoint")) + "P | CJ ONE 포인트 사용";
			}
		}

		$("MyDiscount").innerHTML = disdesc;
		$("MyDiscount").title = disdesc;

		//쿠폰내역 표시
		if(eval($F("Coupon_DiscountAmt")) > 0){
			$("MyCoupon").innerHTML = fnAddComma($F("Coupon_DiscountAmt")) + "원 | " + $F("CouponName");

	function fnPayDiscount(amt, code){
		var retamt = amt;
		var disamt = 0;

		if (code == "34087"){
			if (amt >= 200000){
				disamt = 10000;
			}else if (amt >= 200){
				disamt = amt - fnMoneyTenWonCut(amt * 0.95);
			}
			retamt = amt - disamt;
		}else if (code == "34088"){
			/*
			disamt = amt - fnMoneyTenWonCut(amt - (amt * 0.05));
			if (disamt >= 3000){
				disamt = 3000;
			}else{
				retamt = disamt;
			}
			retamt = amt - disamt;
			*/
			if (amt >= 20000){
				retamt = amt - 1000;
			}else{
				retamt = amt;
			}
		}else if (code == "34092"){
			if (amt >= 200000){
				disamt = 10000;
			}else if (amt >= 10000){
				disamt = fnMoneyTenWonCut(amt * 0.05);
			}
			retamt = amt - disamt;
		}

		return retamt;
	}
